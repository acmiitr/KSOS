105 bool map_page(uint32_t virtual_address,uint32_t physical_address,bool isUser,bool isWritable) //TODO: Need to deal with freed frames && Make this cleaner!!
106 {
107  virtual_address -= (virtual_address%BLOCK_SIZE) ? virtual_address%BLOCK_SIZE : 0;
108  physical_address -= (physical_address%BLOCK_SIZE) ? physical_address%BLOCK_SIZE : 0;
109 
110  uint32_t pd_index = virtual_address >> 22;
111  if (!(_page_directory[pd_index] & PDE_PRESENT))
112  if(!alloc_page(_page_directory+pd_index)) return false;
113 
114  if(isWritable) _page_directory[pd_index] |= PDE_WRITABLE;
115  if(isUser) _page_directory[pd_index]|= PDE_USER;
116 
117  uint32_t* page_table = (uint32_t*)(PAGE_TABLE | (pd_index<<12)); //This is the virtual address of the page table
118  uint32_t pt_index = ((virtual_address >> 12) & 0x3FF); //Using recursive page table technique
119 
120  page_table[pt_index] = physical_address;
121  page_table[pt_index] |= PTE_PRESENT;
122  page_table[pt_index] |= PTE_WRITABLE;
123 
124  if(isUser)
125  {
126  page_table[pt_index]|= PTE_USER;
127  if(isWritable) page_table[pt_index] |= PTE_WRITABLE;
128  else page_table[pt_index] &= ~PTE_WRITABLE;
129  }
130  else
131  {
132  page_table[pt_index] &= ~PTE_USER;
133  page_table[pt_index] |= PTE_WRITABLE; //Writable anyway in kmode
134  }
135 
136  flush_tlb_entry(virtual_address);
137  return true;
138 }
